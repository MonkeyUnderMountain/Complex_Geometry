    \begin{example}\label{eg:Fubini-Study_metric_on_P^n}
        Let \(\mathbb{P}^n\) be the complex projective space.
        The \emph{Fubini-Study metric} is a Hermitian metric on the tautological line bundle \(\mathcal{O}_{\mathbb{P}^n}(-1)\) defined as follows:
        For a point \([z_0: z_1: \cdots : z_n] \in \mathbb{P}^n\) and a vector \(v \in \mathcal{O}_{\mathbb{P}^n}(-1)_{[z]}\), we define
        \[\langle v, v \rangle_{[z]} = \frac{|v|^2}{\sum_{i=0}^n |z_i|^2}. \]

\subsection{the K\"ahler condition}

    \begin{definition}\label{def:Kahler_manifold}
        A \emph{K\"ahler manifold} is a complex manifold \(X\) equipped with a Hermitian metric \(h\) whose associated (1,1)-form \(\omega\) is closed, i.e., \(d\omega = 0\).
